图的遍历

概念

从图中某一个顶点出发访遍图中其余顶点，且使每一个顶点仅被访问一次，这一过程就叫做图的遍历（Traversing Graph）
有两种遍历方案：

  1. 深度优先遍历（Depth First Search），又称为深度优先搜索，简称DFS。
  2. 广度优先遍历（Breadth First Search）,又称为广度优先搜索，简称BFS。

深度优先遍历

深度优先搜索类似于树的先序遍历，具体过程如下：

1.从图中v0出发，访问v0。

2.找出v0的第一个未被访问的邻接点，访问该顶点。以该顶点为新顶点，重复此步骤，直至刚访问过的顶点没有未被访问的邻接点为止。

3.返回前一个访问过的仍有未被访问邻接点的顶点，继续访问该顶点的下一个未被访问领接点。

4.重复2,3步骤，直至所有顶点均被访问，搜索结束。

使用邻接矩阵作为存储结构的代码

#define MAX 100
_Bool visited[MAX];     // 访问标志的数组

/* 邻接矩阵的深度优先递归算法 */
void DFS(MGraph G, int i) {
    int j;
    visited[i] = 1;
    printf("%c ", G.vexes[i]);  // 遍历该结点的操作，也可以是其他操作
    for (j = 0; j < G.numVertexes; ++j) {
        if (G.arcs[i][j] == 1 && !visited[j])
            DFS(G, j);      // 对未访问的邻接顶点递归调用
    }
}

/* 邻接矩阵的深度优先非递归算法 */
void DFS2(MGraph G, int i) {
    int stack[MAXVEX];
    int top = -1;
    int j;
    stack[++top] = i;
    while (top != -1) {
        int k = stack[top--];
        if (!visited[k]) {
            printf("%c ", G.vexes[k]);
            visited[k] = 1;
            for (j = 0; j < G.numVertexes; ++j) {
                if (G.arcs[k][j] == 1 && !visited[j])
                    stack[++top] = j;
            }
        }
    }
}

void DFSTraverse(MGraph G) {
    int i;
    for (i = 0; i < G.numVertexes; ++i) 
        visited[i] = 0;     // 初始化所有顶点状态都是未访问过状态
    for (i = 0; i < G.numVertexes; ++i)
        if (!visited[i])    // 对未访问过的顶点调用DFS，若是连通图，只会执行一次。
            DFS2(G, i);
}

使用邻接表作为存储结构的代码

#define MAX 100
_Bool visited[MAX];     //访问标记的数组

/* 邻接表的深度优先递归算法 */
void DFS(AdjListGraph G, int i) {
    EdgeNode *p;
    visited[i] = 1;
    printf("%c ", G.adjList[i].data);   // 遍历该结点的操作，也可以是其他操作
    p = G.adjList[i].firstedge;
    while (p) {
        if (!visited[p->adjvex])
            DFS(G, p->adjvex);
        p = p->next;
    }
}

/* 邻接表的深度优先非递归算法 */
void DFS2(AdjListGraph G, int i) {
    int stack[MAXVEX];
    int top = -1;
    EdgeNode *p = G.adjList[i].firstedge;
    stack[++top] = i;
    while (top != -1) {
        int k = stack[top--];
        if (!visited[k]) {
            printf("%c ", G.adjList[k].data);
            visited[k] = 1;
            p = G.adjList[k].firstedge;
            while (p) {
                if (!visited[p->adjvex])
                    stack[++top] = p->adjvex;
                p = p->next;
            }
        }
    }
}

void DFSTraverse(AdjListGraph G) {
    int i;
    for (i = 0; i < G.numVertexes; ++i)
        visited[i] = 0; // 初始化所有顶点状态都是未访问过状态
    for (i = 0; i < G.numVertexes; ++i)
        if (!visited[i])
            DFS2(G, i);
}

对于n个顶点e条边的图来说，邻接矩阵遍历的时间复杂度为O(n²)，邻接表遍历的时间复杂度为O(n+e)。显然对于稀疏图来说，邻接表结构的算法在时间效率上大大提高。

广度优先遍历

广度优先搜索类似于树的层次遍历过程。它需要借助一个队列来实现。
只要队列不空，则重复如下操作：

1.队头顶点u出队。

2.依次检查u的所有邻接顶点w，若w未被访问，则访问w，并将visited[w]置为true，同时将w入队。

使用邻接矩阵作为存储结构的代码

#define MAX 100
_Bool visited[MAXVEX];

void BFS(MGraph G, int i) {
    int queue[MAX];
    int front = 0, rear = 0;
    queue[rear] = i;
    rear = (rear+1) % MAX;
    printf("%c ", G.vexes[i]);
    visited[i] = 1;
    while (rear != front) {
        int j;
        int k = queue[front];
        front = (front+1) % MAX;
        for (j = 0; j < G.numVertexes; ++j)
            if (G.arcs[k][j] == 1 && !visited[j]) {
                queue[rear] = j;
                rear = (rear+1) % MAX;
                printf("%c ", G.vexes[j]);
                visited[j] = 1;                    
            }
    }
}

void BFSTraverse(MGraph G) {
    int i;
    for (i = 0; i < G.numVertexes; ++i)
        visited[i] = 0;
    for (i = 0; i < G.numVertexes; ++i)
        if (!visited[i])
            BFS(G, i);
}

使用邻接表作为存储结构的代码

#define MAX 100
_Bool visited[MAX];     //访问标记的数组

void BFS(AdjListGraph G, int i) {
    int queue[MAX];
    int front = 0, rear = 0;
    queue[rear] = i;
    rear = (rear+1) % MAX;
    printf("%c ", G.adjList[i].data);
    visited[i] = 1;
    while (front != rear) {
        int k = queue[front];
        front = (front+1) % MAX;
        EdgeNode *p = G.adjList[k].firstedge;
        while (p) {
            if (!visited[p->adjvex]) {
                printf("%c ", G.adjList[p->adjvex].data);
                visited[p->adjvex] = 1;
                queue[rear] = p->adjvex;
                rear = (rear+1) % MAX;
            }
            p = p->next;
        }
    }
}

void BFSTraverse(AdjListGraph G) {
    int i;
    for (i = 0; i < G.numVertexes; ++i)
        visited[i] = 0;
    for (i = 0; i < G.numVertexes; ++i)
        if (!visited[i])
            BFS(G, i);
}

对比

图的深度优先遍历和广度优先遍历在时间复杂度上是一样的，不同之处仅仅在于对顶点访问的顺序不同。二者在全图遍历上没有优劣之分。
应视不同的情况选择不同的算法，如果图的顶点和边非常多，不能再短时间内遍历完成，遍历的目的是为了寻找合适的顶点，此时，深度优先更适合目标比较明确，以找到目标为主要目的的情况，而广度优先更适合再不断扩大遍历范围时找到相对最优解的情况。
